# start app locally

changed start script in package.json
from:
"start": "react-scripts start"
to:
"start": "react-scripts start --host 0.0.0.0"

##npm install

##npm run start

instead of using localhost:3000 use {YOUR_IP}:3000
example: 124.169.2.137:3000
